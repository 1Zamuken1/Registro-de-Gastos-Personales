Guía completa para inicializar el chatbot
1. Descargar Node.js

Ir a https://nodejs.org/
Descargar la versión LTS
Instalar siguiendo el asistente

2. Crear estructura del proyecto
mkdir mi-chatbot
cd mi-chatbot
3. Crear package.json
json{
  "name": "chatbot-local",
  "version": "1.0.0",
  "main": "server.js",
  "scripts": {
    "start": "node server.js"
  },
  "dependencies": {
    "dotenv": "^16.5.0",
    "express": "^5.1.0",
    "node-fetch": "^2.7.0"
  }
}
4. Instalar dependencias
npm install express node-fetch dotenv
npm install
5. Crear archivo .env
OPENAI_API_KEY=tu_clave_de_openrouter_aqui
6. Crear archivo server.js
javascriptrequire("dotenv").config();
const express = require("express");
const fetch = require("node-fetch");
const app = express();

app.use(express.static("public"));
app.use(express.json());

app.post("/chat", async (req, res) => {
  const userMessage = req.body.message;

  try {
    const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${process.env.OPENAI_API_KEY}`,
        "HTTP-Referer": "http://localhost:5500",
        "X-Title": "ChatBot Local"
      },
      body: JSON.stringify({
        model: "mistralai/mistral-7b-instruct",
        messages: [{ role: "user", content: userMessage }]
      }),
    });

    const data = await response.json();

    if (!response.ok) {
      return res.status(500).json({
        reply: data.error?.message || "Error desde OpenRouter",
      });
    }

    const reply = data.choices?.[0]?.message?.content || "Sin respuesta de la IA.";
    res.json({ reply });

  } catch (err) {
    console.error("Error general:", err);
    res.status(500).json({ reply: "Error general del servidor" });
  }
});

const PORT = 5500;
app.listen(PORT, () => {
  console.log(`Servidor corriendo en http://localhost:${PORT}`);
});
7. Crear carpeta public
mkdir public

Ir a https://openrouter.ai/
